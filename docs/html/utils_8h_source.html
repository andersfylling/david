208 constexpr ::std::array<uint64_t, 64> compileKnightAttacks() {
209  ::std::array<uint64_t, 64> attacks{};
210 
211  // directions a rook can move
212  const std::array<int8_t, 4> directions = {1, -1};
213 
214  for (uint8_t i = 0; i < 64; i++) {
215 
216  // the different board limits related to directions
217  uint8_t row = i / 8;
218  uint8_t col = i % 8;
219  //const std::array<uint8_t, 4> limitations = {/*up*/7 - row, /*right*/col, /*down*/row, /*left*/7 - col};
220 
221  uint64_t result = 0ULL;
222 
223  for (uint8_t ii = 0; ii < 2; ii++) {
224  const int8_t directionY = directions[ii];
225  // up / down
226  for (uint8_t y = 1; y < 3; y++) {
227  // make sure the knight doesnt go too high or too "low"
228  const auto heightPush = row + (directionY * y);
229  if (heightPush > 7 || heightPush < 0) {
230  continue;
231  }
232 
233  // how far to the side should the knight move, based on its desired height move
234  uint8_t colOffset = 3 - y; // if y == 1, then move 2 pos to the side, y == 2 move 1 pos to the side
235 
236  for (uint8_t x = 0; x < 2; x++) {
237  const int8_t directionX = directions[x];
238  // make sure it doesn't go too far to the sides
239  const auto sidePush = col + (directionX * colOffset);
240  if (sidePush > 7 || sidePush < 0) {
241  continue;
242  }
243 
244  // add it to the possible moves
245  result |= ::utils::indexToBitboard(i + (directionX * colOffset) + (directionY * 8 * y));
246  }
247  }
248  }
249 
250  attacks[i] = result;
251  }
252 
253  return attacks;
254 };
