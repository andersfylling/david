400  template<uint8_t DIRECTION>
401  inline type::bitboard_t extractLegalRange(
402  const type::bitboard_t south,
403  const type::bitboard_t north,
404  const type::bitboard_t psuedo,
405  const type::bitboard_t friendly,
406  const type::bitboard_t hostiles
407  ) const {
408  using bb = type::bitboard_t;
409 
410  // south blockers
411  bb southBlockers = south & (((psuedo & hostiles) >> DIRECTION) | friendly);
412 
413  // north blockers
414  bb northBlockers = north & (((psuedo & hostiles) << DIRECTION) | friendly);
415 
416  // legal north area
417  bb legalNorth = northBlockers > 0 ? (::utils::indexToBitboard(::utils::LSB(northBlockers & psuedo)) - 1) : 0ull - 1;
418 
419  // illegal south area
420  bb illegalSouth = southBlockers > 0 ? (::utils::indexToBitboard(::utils::MSB(southBlockers & psuedo) + 1) - 1) : 0ull;
421 
422  // legal range
423  return psuedo & (legalNorth ^ illegalSouth);
424  }

443  inline type::bitboard_t generatePawnPaths (const uint8_t index, const type::gameState_t& gs, const bool hostile = false) {
444  const auto pieceBoard = ::utils::indexToBitboard(index);
445  const auto hostiles = gs.piecess[hostile ? 0 : 1];
446  //const auto friendly = gs.piecess[hostile ? 1 : 0];
447  const auto allowedSquares = (~this->state.combinedPieces) ^ ::utils::indexToBitboard(index);
448  const bool whiteMoving = gs.isWhite && !hostile;
449 
450  type::bitboard_t board = 0ULL;
451  type::bitboard_t attacks = ::utils::constant::pawnAttackPaths[index];
452 
453  if (whiteMoving) {
454  // move one and two squares up if possible
455  board = ((pieceBoard << 8) | ((65280ULL & pieceBoard) << 16)) & (allowedSquares & (allowedSquares << 8));
456 
457  // attacks
458  attacks &= (hostiles | ::utils::indexToBitboard(this->state.enPassant));
459  board |= attacks & ~(pieceBoard - 1);
460  }
461 
462  // black
463  else {
464  // move one and two squares up if possible
465  board = ((pieceBoard >> 8) | ((71776119061217280ULL & pieceBoard) >> 16)) & (allowedSquares & (allowedSquares >> 8));
466 
467  // attacks
468  attacks &= (hostiles | (this->state.enPassant > 0 ? ::utils::indexToBitboard(this->state.enPassant) : 0)); // hostile pieces that can be attacked.
469  board |= attacks & (pieceBoard - 1);
470  }
471 
472  // extract the promotions from the moves
473  type::bitboard_t promotions = 18374686479671623935ULL & board;
474  board ^= promotions;
475 
476  while (promotions != 0) {
477  unsigned int i = ::utils::LSB(promotions);
478  promotions = ::utils::flipBitOffCopy(promotions, i);
479 
480  this->generatePromotions(::utils::indexToBitboard(i), pieceBoard);
481  }
482 
483  return board;
484  }

542  inline type::bitboard_t generateDiagonals (const uint8_t index, const type::gameState_t& gs, const bool hostilePath = false) const {
543  const auto friendly = gs.piecess[hostilePath ? 1 : 0];
544  const auto hostiles = gs.piecess[hostilePath ? 0 : 1];
545  const auto iBoard = ::utils::indexToBitboard(index);
546  const type::bitboard_t southArea = iBoard - 1;
547  const type::bitboard_t northArea = (~iBoard) ^ southArea;
548 
549  // attack path starting from bottom going upwards, reading left to right
550  type::bitboard_t r1 = ::utils::constant::diagonalDUAttackPaths[index];
551 
552  r1 = this->extractLegalSouthPath(r1, friendly, hostiles, southArea & r1, 7);
553  r1 = this->extractLegalNorthPath(r1, friendly, hostiles, northArea & r1, 7);
554 
555  // attack path starting from bottom going upwards, reading left to right
556  type::bitboard_t r2 = ::utils::constant::diagonalUDAttackPaths[index];
557 
558  r2 = this->extractLegalSouthPath(r2, friendly, hostiles, southArea & r2, 9);
559  r2 = this->extractLegalNorthPath(r2, friendly, hostiles, northArea & r2, 9);
560 
561  // combine
562  return r1 | r2; // remove start position of piece
